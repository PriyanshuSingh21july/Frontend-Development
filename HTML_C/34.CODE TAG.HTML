class Solution { // Function to find floor of x // arr: input array // n is the size of array static int findFloor(long arr[], int n, long x) { for (int i = n - 1; i >= 0; i--) { if (arr[i] < x) { return i; } } // Return -1 only if none of the elements is less than x return -1; } }
